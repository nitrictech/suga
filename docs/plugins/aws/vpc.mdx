---
title: VPC
description: Virtual Private Cloud with isolated network and subnets
---

## Configuration

<ParamField path="azs" type="list(string)">
  List of availability zones to use for the VPC
  
  **Example:** `["us-west-2a", "us-west-2b", "us-west-2c"]`
</ParamField>

<ParamField path="default_security_group_egress" type="list(map(string))">
  Egress rules for the default security group
  
  **Example:** `[{"from_port": "0", "to_port": "0", "protocol": "-1", "cidr_blocks": "0.0.0.0/0"}]`
</ParamField>

<ParamField path="default_security_group_ingress" type="list(map(string))">
  Ingress rules for the default security group
  
  **Example:** `[{"from_port": "443", "to_port": "443", "protocol": "tcp", "cidr_blocks": "10.0.0.0/16"}]`
</ParamField>

<ParamField path="enable_nat_gateway" type="bool" default="false">
  Enable NAT Gateway for private subnet internet access
</ParamField>

<ParamField path="enable_vpn_gateway" type="bool" default="false">
  Enable VPN Gateway for the VPC
</ParamField>

<ParamField path="name" type="string" required>
  The name of the VPC
  
  **Example:** `main-vpc`
</ParamField>

<ParamField path="networking" type="object">
  Network configuration including CIDR block and subnet definitions
  
  **Example:** `{"cidr_block": "10.0.0.0/16", "private_subnets": ["10.0.1.0/24", "10.0.2.0/24", "10.0.3.0/24"], "public_subnets": ["10.0.101.0/24", "10.0.102.0/24", "10.0.103.0/24"]}`
</ParamField>

<ParamField path="single_nat_gateway" type="bool" default="false">
  Use a single NAT Gateway instead of one per availability zone
</ParamField>

<ParamField path="tags" type="map(string)">
  Tags to apply to the VPC resources
  
  **Example:** `{"Environment": "production", "Team": "platform"}`
</ParamField>

## Outputs

<ResponseField name="default_security_group_id" type="string">
  The ID of the default security group
  
  **Example:** `sg-0123456789abcdef0`
</ResponseField>

<ResponseField name="private_subnets" type="list(string)">
  List of private subnet IDs
  
  **Example:** `["subnet-12345678", "subnet-23456789"]`
</ResponseField>

<ResponseField name="public_subnets" type="list(string)">
  List of public subnet IDs
  
  **Example:** `["subnet-87654321", "subnet-76543210"]`
</ResponseField>

<ResponseField name="subnets" type="list(string)">
  List of all subnet IDs (public and private)
  
  **Example:** `["subnet-12345678", "subnet-23456789", "subnet-87654321", "subnet-76543210"]`
</ResponseField>

<ResponseField name="vpc_id" type="string">
  The ID of the VPC
  
  **Example:** `vpc-12345678`
</ResponseField>
